// MYSQL SCRIPT : 
CREATE DATABASE box_share;

use box_share;

CREATE TABLE Box (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL
);

CREATE TABLE File (
    id INT AUTO_INCREMENT PRIMARY KEY,
    filename VARCHAR(255) NOT NULL,
    box_id INT,
    FOREIGN KEY (box_id) REFERENCES Box(id)
);



COMMIT;
